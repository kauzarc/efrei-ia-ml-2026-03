\section{Du contexte à l'attention}
% ============================================================

\begin{frame}{Le problème du contexte}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{alertblock}{La polysémie en pratique}
        \begin{itemize}
          \item «~La \textbf{souris} mange le fromage.~»
          \item «~Ma \textbf{souris} d'ordinateur est cassée.~»
        \end{itemize}
        \vspace{4pt}
        Dans Word2Vec, «~souris~» n'a \textbf{qu'un seul vecteur}
        --- une moyenne des deux sens.
      \end{alertblock}
      \smallskip
      \begin{block}{Les autres limites (rappel J4-AM)}
        \begin{itemize}
          \item Embeddings \textbf{figés} après entraînement
          \item Ignorent l'\textbf{ordre} des mots dans la phrase
          \item Biais hérités du corpus d'entraînement
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{exampleblock}{Le besoin : embeddings contextuels}
        Le vecteur d'un mot doit être calculé
        \textbf{à la volée} en fonction de ses voisins.
        \vspace{4pt}
        \begin{center}
          \begin{tikzpicture}[font=\tiny,
            box/.style={draw, rounded corners=3pt, align=center,
                        minimum width=3.0cm, minimum height=0.55cm}]
            \node[box, fill=jedy_example!15, draw=jedy_example] (b1) at (0,0)
              {«~La souris \textbf{mange}\ldots~»};
            \node[box, fill=jedy_blue!15, draw=jedy_blue] (b2) at (0,-0.95)
              {«~Ma souris d'\textbf{ordinateur}\ldots~»};
            \draw[->, jedy_example, thick] (b1.east) -- (2.1, 0.0)
              node[right] {$\vec{v}_{\text{animal}}$};
            \draw[->, jedy_blue, thick] (b2.east) -- (2.1,-0.95)
              node[right] {$\vec{v}_{\text{outil}}$};
          \end{tikzpicture}
        \end{center}
      \end{exampleblock}
      \vspace{2pt}
      \begin{alertblock}{Solution}
        Le \textbf{mécanisme d'attention} --- cœur du Transformer.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{L'intuition de l'attention}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{exampleblock}{Principe \& exemple}
        Chaque mot «~vote~» pour ceux qui l'aident à se définir.\\[2pt]
        «~L'animal \ldots\ car \textbf{il} était fatigué.~»\\[2pt]
        Distribution d'attention de «~il~» :
        \begin{itemize}
          \item \textbf{animal} : \textbf{90\,\%}
          \item fatigué : 6\,\%
          \item autres mots : 4\,\%
        \end{itemize}
      \end{exampleblock}
      \vspace{2pt}
      \begin{alertblock}{Avantage : parallélisme}
        Contrairement aux RNN, l'attention traite
        \textbf{tous les mots simultanément} --- GPU.
      \end{alertblock}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Distribution d'attention de «~\textbf{il}~»}
        \begin{itemize}
          \item[\textcolor{jedy_alert}{\textbf{90\,\%}}]
                \textbf{animal} --- co-référence directe
          \item[\textcolor{jedy_mid}{6\,\%}]
                \textbf{fatigué} --- trait partagé
          \item[4\,\%] autres mots (négligeable)
        \end{itemize}
      \end{block}
      \vspace{2pt}
      \begin{exampleblock}{Interprétation}
        Le modèle «~sait~» que «~il~» réfère à «~animal~»
        \textbf{grâce au contexte} global --- sans règle
        grammaticale explicite.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{La formule de l'attention}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Mécanisme Query / Key / Value}
        \begin{itemize}
          \item $Q$ (\textit{Query}) : «~Ce que je \textbf{cherche}.~»
          \item $K$ (\textit{Key}) : «~Ce que j'\textbf{offre}.~»
          \item $V$ (\textit{Value}) : «~L'information que je \textbf{contiens}.~»
        \end{itemize}
      \end{block}
      \smallskip
      \begin{block}{Équation}
        \[
          \mathrm{Attention}(Q,K,V)
          = \mathrm{softmax}\!\left(\frac{QK^{\T}}{\sqrt{d_k}}\right)V
        \]
        \begin{itemize}
          \item $QK^{\T}$ : score de pertinence (mot, mot)
          \item $\sqrt{d_k}$ : normalisation (gradients stables)
          \item Softmax $\to$ poids $\in [0,1]$, somme $= 1$
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Multi-Head Attention}
        On lance $h$ attentions en \textbf{parallèle}
        sur des sous-espaces distincts :
        \begin{itemize}
          \item Tête 1 : relations \textbf{grammaticales}
          \item Tête 2 : relations \textbf{sémantiques}
          \item Tête 3 : relations de \textbf{coréférence}
          \item \ldots
        \end{itemize}
        \vspace{4pt}
        Les sorties sont \textbf{concaténées} puis projetées.
      \end{exampleblock}
      \vspace{2pt}
      \begin{alertblock}{Note «~Geek~»}
        C'est une \textbf{recherche floue} dans une base de données
        vectorielle : $Q$ est la requête, $K$ l'index, $V$ le résultat.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

  \begin{columns}[T]
    \begin{column}{0.31\textwidth}
      \begin{block}{Encoder (ex: BERT)}
        \begin{itemize}
          \item[\textcolor{jedy_example}{$+$}] Lit la phrase
                \textbf{dans les deux sens}
          \item[\textcolor{jedy_example}{$+$}] Embeddings contextuels

          \item Idéal : \textbf{classification},

        \begin{itemize}
          \item[\textcolor{jedy_example}{$+$}] Génère mot après mot

          \item Attention \textbf{masquée} : ne voit que le passé

        \begin{itemize}
          \item[\textcolor{jedy_example}{$+$}] Encode la source,

          \item Idéal : \textbf{traduction},

\begin{frame}{Le paradigme du Transfer Learning}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{alertblock}{Pre-training --- la fondation}
        \begin{itemize}
          \item Entraînement massif sur \textbf{tout Internet}
                (Wikipédia, livres, code\ldots)
          \item Objectif : \textbf{prédire le token masqué} (BERT)
                ou le token suivant (GPT)
          \item Coût : millions d'euros de calcul GPU
          \item Résultat : le modèle apprend la \textbf{«~langue~»}
                --- grammaire, faits, raisonnement
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{exampleblock}{Fine-tuning --- l'adaptation}
        On ajuste le modèle sur une \textbf{tâche métier} :
        \begin{itemize}
          \item Classification de tickets support
          \item Résumé automatique de rapports
          \item Détection de sentiment client
        \end{itemize}
        \vspace{4pt}
        \textbf{Quelques dizaines de lignes} de code,
        quelques \textbf{minutes} de calcul $\to$
        niveau état de l'art.
      \end{exampleblock}
      \smallskip
      \begin{block}{Pourquoi ça marche ?}
        Les couches inférieures encodent des
        connaissances \textbf{générales} réutilisables ;
        seule la tête de classification change.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{L'écosystème Hugging Face}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{The Hub}
        Le «~GitHub~» des modèles d'IA :
        \begin{itemize}
          \item $> 100\,000$ modèles prêts à l'emploi
          \item Hébergement gratuit des poids
          \item Versionning, fiches modèles,
                benchmarks intégrés
        \end{itemize}
      \end{block}
      \smallskip
      \begin{block}{\texttt{datasets}}
        \begin{itemize}
          \item Accès instantané à des milliers de
                bases de données (IMDb, Wikipedia,
                GLUE, SQuAD\ldots)
          \item Chargement en une ligne :
          \item[] \quad\texttt{\small load\_dataset("imdb")}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{\texttt{transformers}}
        Interface \textbf{unifiée} pour tout modèle :
        \begin{itemize}
          \item BERT, RoBERTa, CamemBERT
          \item GPT-2, LLaMA, Mistral
          \item T5, BART, Whisper, CLIP\ldots
        \end{itemize}
        Même API quel que soit l'architecture.
      \end{exampleblock}
      \smallskip
      \begin{alertblock}{Écosystème complet}
        \begin{itemize}
          \item \texttt{tokenizers} : tokeniseurs rapides (Rust)
          \item \texttt{evaluate} : métriques standardisées
          \item \texttt{accelerate} : multi-GPU/TPU
          \item \texttt{PEFT} : fine-tuning efficace (LoRA)
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{La \texttt{pipeline} API --- le mode «~Product~»}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Concept}
        Abstraction totale : téléchargement du modèle,

from transformers import pipeline

clf = pipeline("sentiment-analysis")

\end{lstlisting}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Tâches disponibles}

          \item \texttt{"sentiment-analysis"}

          \item \texttt{"summarization"}
          \item \texttt{"translation"}
          \item \texttt{"question-answering"}
          \item \texttt{"ner"} (entités nommées)

      \end{exampleblock}

      \begin{alertblock}{Intérêt terrain}
        C'est ce qu'un dev Fullstack utilisera

\begin{frame}[fragile]{La tokenisation en pratique}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{block}{Sous-mots (BPE / WordPiece)}
        Les tokenizers modernes découpent les mots rares en
        \textbf{sous-unités} pour couvrir toute la langue
        avec un vocabulaire fixe ($\approx 30\,000$ tokens) :
        \begin{itemize}
          \item \texttt{"tokenisation"} $\to$
                \texttt{["token",\ "\#\#isation"]}
          \item \texttt{"unhappiness"} $\to$
                \texttt{["un",\ "\#\#happy",\ "\#\#ness"]}
        \end{itemize}
      \end{block}
      \smallskip
      \begin{alertblock}{Paramètres critiques}
        \begin{itemize}
          \item \texttt{padding} : aligne les séquences du batch
          \item \texttt{truncation} : limite à \texttt{max\_length}
          \item \texttt{attention\_mask} : 1~=~vrai token,~0~=~padding
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.46\textwidth}
\begin{lstlisting}[language=Python]
from transformers import AutoTokenizer

tok = AutoTokenizer.from_pretrained(
    "camembert-base")

batch = tok(
    ["Bonjour",
     "Je suis etudiant en M2"],
    padding=True,
    truncation=True,
    max_length=128,
    return_tensors="pt",
)
# batch["input_ids"]       [2, L]
# batch["attention_mask"]  [2, L]
# L = longueur de la plus longue phrase
\end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Anatomie du Fine-Tuning --- le mode «~Expert~»}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Les 3 composants obligatoires}
        \begin{enumerate}
          \item \textbf{Tokenizer} : identique au modèle pré-entraîné
          \item \textbf{Model Head} : couche de sortie remplacée
                (ex: $30\,000 \to 2$ classes)
          \item \textbf{Trainer} : boucle d'entraînement optimisée
        \end{enumerate}
      \end{block}
      \vspace{2pt}
\begin{lstlisting}[language=Python]
from transformers import (
    AutoTokenizer,
    AutoModelForSequenceClassification,
)
model = AutoModelForSequenceClassification\
  .from_pretrained("camembert-base",
                   num_labels=2)
tokenizer = AutoTokenizer\
  .from_pretrained("camembert-base")
\end{lstlisting}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Pourquoi le Tokenizer doit coïncider ?}
        Le modèle a appris des embeddings pour des IDs
        précis. Si l'on utilise un tokenizer différent,
        les IDs ne correspondent plus aux poids ---
        le modèle produit du bruit.
      \end{exampleblock}
      \vspace{2pt}
      \begin{block}{Alternatives légères}
        \begin{itemize}
          \item \textbf{LoRA / QLoRA} : $< 1\,\%$ des params
          \item \textbf{Prompt-tuning} : tokens «~soft~»
          \item \textbf{Zero-shot} : sans fine-tuning
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Le Trainer --- boucle complète}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
\begin{lstlisting}[language=Python, basicstyle=\tiny\ttfamily]
from transformers import (
    TrainingArguments, Trainer,
    DataCollatorWithPadding)

# dataset tokenise via .map() (cf. slide precedente)

args = TrainingArguments(
    output_dir="./results",
    num_train_epochs=3,
    per_device_train_batch_size=16,
    eval_strategy="epoch",
    load_best_model_at_end=True)

trainer = Trainer(
    model=model,
    args=args,
    train_dataset=ds["train"],
    eval_dataset=ds["test"],
    data_collator=DataCollatorWithPadding(
        tokenizer))

trainer.train()
\end{lstlisting}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Ce que gère le Trainer}
        \begin{itemize}
          \item Shuffling, batching, padding dynamique
          \item Gradient accumulation
          \item Checkpoints + évaluation automatiques
        \end{itemize}
      \end{block}
      \vspace{2pt}
      \begin{exampleblock}{Pour aller plus loin}
        \begin{itemize}
          \item \texttt{compute\_metrics} : F1 à chaque epoch
          \item \texttt{callbacks} : early stopping
          \item \texttt{accelerate} : multi-GPU
        \end{itemize}
      \end{exampleblock}
      \vspace{2pt}
      \begin{alertblock}{Séquence typique}
        \texttt{map()} $\to$ \texttt{Trainer()} $\to$
        \texttt{train()} $\to$ \texttt{push\_to\_hub()}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Vers les LLM modernes}
% ============================================================

\begin{frame}{Instruction Tuning \& Alignement (RLHF)}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{alertblock}{Le problème du modèle de base}
        Un LLM pré-entraîné complète du texte brut ;
        il ne \textbf{suit pas les instructions} :
        \begin{itemize}
          \item Entrée : «~Explique la photosynthèse.~»
          \item Sortie : «~Explique la photosynthèse.
                Exercice 3\ldots~» (continuation !)
        \end{itemize}
      \end{alertblock}
      \smallskip
      \begin{block}{Pipeline d'alignement}
        \begin{enumerate}
          \item \textbf{SFT} : fine-tuning supervisé sur des paires
                question / réponse rédigées par des humains
          \item \textbf{Reward Model} : apprend à préférer
                une bonne réponse sur une mauvaise
          \item \textbf{RLHF} : le LLM est optimisé pour
                maximiser la récompense humaine
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{exampleblock}{Résultat}
        \begin{itemize}
          \item[\textcolor{jedy_example}{$+$}]
                Suit les instructions précisément
          \item[\textcolor{jedy_example}{$+$}]
                Refuse les requêtes dangereuses
          \item[\textcolor{jedy_example}{$+$}]
                Répond de façon utile et concise
        \end{itemize}
        \vspace{4pt}
        C'est ce qui transforme GPT-2 en ChatGPT,
        LLaMA en LLaMA-Chat / Mistral-Instruct.
      \end{exampleblock}
      \vspace{4pt}
      \begin{block}{Variantes modernes}
        \begin{itemize}
          \item \textbf{DPO} : plus stable que RLHF classique
          \item \textbf{Constitutional AI} (Anthropic) :
                l'IA se critique elle-même
          \item \textbf{RLAIF} : feedback automatique,
                sans annotation humaine
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Prompt Engineering}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Zero-shot vs Few-shot}
        \textbf{Zero-shot} : instruction seule, sans exemple.\\[2pt]
        \textbf{Few-shot} : 2--5 exemples dans le prompt.
      \end{block}
\begin{lstlisting}[language=Python, basicstyle=\tiny\ttfamily]
# Zero-shot
prompt = """Classe ce tweet :
POSITIF ou NEGATIF.
Tweet: "Ce cours est top !"
Sentiment:"""

# Few-shot
prompt = """Tweet: "J'adore !" -> POSITIF
Tweet: "Nul !"   -> NEGATIF
Tweet: "Sympa mais long." ->"""
\end{lstlisting}
      \begin{alertblock}{Chain-of-Thought (CoT)}
        Ajouter «~Raisonne étape par étape~»
        améliore fortement les tâches logiques.
      \end{alertblock}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Quand fine-tuner plutôt que prompter ?}
        \begin{itemize}
          \item Vocabulaire très spécialisé
                (médical, légal, code propriétaire)
          \item Fenêtre de contexte insuffisante
          \item Coût d'inférence à réduire
          \item Données confidentielles
                (pas d'API externe)
        \end{itemize}
      \end{block}
      \smallskip
      \begin{exampleblock}{Règle pratique}
        \textbf{Commencer par le prompting.}\\[2pt]
        Fine-tuner si les limites sont atteintes ---
        coût en temps, données et argent.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{RAG --- Retrieval-Augmented Generation}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{alertblock}{Limites des LLM seuls}
        \begin{itemize}
          \item Connaissance figée à la date d'entraînement
          \item Hallucinent des faits non vus
          \item Ne connaissent pas vos données internes
                (docs techniques, base clients\ldots)
        \end{itemize}
      \end{alertblock}
      \smallskip
      \begin{exampleblock}{Solution : RAG}
        \begin{enumerate}
          \item \textbf{Indexer} les documents en embeddings
                dans une base vectorielle
          \item À chaque question : \textbf{chercher} les $k$
                passages les plus proches
          \item \textbf{Injecter} ces passages dans le prompt
          \item Le LLM \textbf{répond} en se basant sur
                des sources vérifiées
        \end{enumerate}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.46\textwidth}
      \vspace{-4pt}
      \begin{center}
        \begin{tikzpicture}[font=\scriptsize,
          blk/.style={draw, rounded corners=3pt, align=center,
                      minimum width=3.0cm, minimum height=0.52cm},
          arr/.style={->, thick, gray!60}]
          \node[blk, fill=jedy_blue!12, draw=jedy_blue]
            (q)   at (0, 0)    {Question utilisateur};
          \node[blk, fill=jedy_example!12, draw=jedy_example]
            (db)  at (0,-1.0)  {Base vectorielle};
          \node[blk, fill=jedy_blue!12, draw=jedy_blue]
            (ret) at (0,-2.0)  {Top-$k$ passages};
          \node[blk, fill=jedy_alert!12, draw=jedy_alert]
            (llm) at (0,-3.0)  {LLM + prompt enrichi};
          \node[blk, fill=jedy_example!12, draw=jedy_example]
            (out) at (0,-4.0)  {Réponse sourcée};
          \draw[arr] (q)   -- (db);
          \draw[arr] (db)  -- (ret);
          \draw[arr] (ret) -- (llm);
          \draw[arr] (llm) -- (out);
          \draw[->, dashed, gray!50]
            (q.east) -- ++(0.7,0)
            -- ++(0,-3.0) -- (llm.east);
        \end{tikzpicture}
      \end{center}

        LangChain, LlamaIndex, Chroma, FAISS,
        pgvector (PostgreSQL)
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Évaluation : au-delà de l'accuracy}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{alertblock}{Pourquoi l'accuracy ment}
        99\,\% de mails normaux, 1\,\% de spams.\\[2pt]
        Prédire «~normal~» partout $\to$ \textbf{99\,\%} ---
        mais le modèle est \textbf{inutile}.
      \end{alertblock}
      \smallskip
      \begin{block}{Précision \& Rappel}
        \vspace{2pt}
        $\displaystyle\text{Précision} = \dfrac{TP}{TP{+}FP}$
        \qquad
        $\displaystyle\text{Rappel} = \dfrac{TP}{TP{+}FN}$
        \vspace{2pt}
      \end{block}
      \smallskip
      \begin{exampleblock}{F1-Score --- moyenne harmonique}
        \[
          F_1 = 2 \cdot
          \frac{\text{Précision} \times \text{Rappel}}
               {\text{Précision} + \text{Rappel}}
        \]
      \end{exampleblock}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Matrice de confusion}
        Visualiser \textbf{où} le modèle se trompe :
        \vspace{2pt}
        \begin{center}
          \begin{tikzpicture}[font=\scriptsize]
            \node[draw, fill=jedy_example!25,
                  minimum width=1.5cm, minimum height=0.75cm, align=center,
                  font=\scriptsize] at (0.75, -0.375)  {TP\\{\tiny vrais +}};
            \node[draw, fill=jedy_alert!25,
                  minimum width=1.5cm, minimum height=0.75cm, align=center,
                  font=\scriptsize] at (2.25, -0.375)  {FP\\{\tiny faux +}};
            \node[draw, fill=jedy_alert!25,
                  minimum width=1.5cm, minimum height=0.75cm, align=center,
                  font=\scriptsize] at (0.75, -1.125)  {FN\\{\tiny faux --}};
            \node[draw, fill=jedy_example!25,
                  minimum width=1.5cm, minimum height=0.75cm, align=center,
                  font=\scriptsize] at (2.25, -1.125)  {TN\\{\tiny vrais --}};
            \node[font=\tiny, color=gray] at (0.75,  0.1)  {Prédit +};
            \node[font=\tiny, color=gray] at (2.25,  0.1)  {Prédit --};

\end{frame}

\begin{frame}{Métriques de génération : BLEU \& ROUGE}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{BLEU --- pour la traduction}
        Mesure la \textbf{précision} : quelle fraction des
        n-grammes produits apparaît dans la référence ?
        \vspace{2pt}
        \begin{itemize}
          \item Compare les \textbf{n-grammes} produits
                à la traduction de référence
          \item BP : pénalité si la sortie est trop courte
          \item $\in [0,1]$ ; $> 0{,}3$ considéré bon
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{ROUGE --- pour le résumé}
        Mesure le \textbf{rappel} : quelle fraction des
        n-grammes de la référence est retrouvée ?
        \begin{itemize}
          \item \textbf{ROUGE-1} : unigrammes communs
          \item \textbf{ROUGE-2} : bigrammes communs
          \item \textbf{ROUGE-L} : plus longue sous-séquence
        \end{itemize}
      \end{block}
      \smallskip
      \begin{alertblock}{Limite commune}
        BLEU et ROUGE sont \textbf{lexicaux} --- ils ratent
        les reformulations correctes sémantiquement.
        \textbf{BERTScore} (similarité cosinus d'embeddings)
        est plus robuste pour évaluer la qualité réelle.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}
