\begin{frame}{Le défi du langage naturel}
  \begin{columns}[T]

        \begin{itemize}
          \item Image : grille de \textbf{pixels} $=$ nombres réels $\in [0,1]$
          \item Texte : suite de \textbf{symboles discrets}
          \item Comment calculer la «~dérivée~» d'un mot ?
          \item Comment gérer la \textbf{polysémie} ?

      \end{alertblock}
      \smallskip
      \begin{exampleblock}{Objectif du matin}
        Construire un \textbf{pont} entre le monde discret (mots)

\begin{frame}{Tokenisation (1/2) --- Approches naïves}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Niveau caractère}
        \begin{itemize}
          \item[\textcolor{jedy_example}{$+$}] Petit vocabulaire ($\sim$256)
          \item[\textcolor{jedy_example}{$+$}] Aucun mot inconnu (OOV)
          \item[\textcolor{jedy_alert}{$-$}] Perte de la sémantique locale
          \item[\textcolor{jedy_alert}{$-$}] Séquences très longues
        \end{itemize}
        \smallskip
        \code{chat} $\;\to\;$ \texttt{\small[c, h, a, t]}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Niveau mot (\textit{word-level})}
        \begin{itemize}
          \item[\textcolor{jedy_example}{$+$}] Sens clair, séquences courtes
          \item[\textcolor{jedy_alert}{$-$}] Vocabulaire immense ($V > 100\,000$)
          \item[\textcolor{jedy_alert}{$-$}] Mots hors dictionnaire (OOV)
          \item[\textcolor{jedy_alert}{$-$}] Flexions = tokens distincts
        \end{itemize}
        \smallskip
        «~mangeras~» $\ne$ «~mangeons~»
      \end{block}
    \end{column}
  \end{columns}
  \bigskip
  \begin{alertblock}{Le dilemme}
    Caractère : trop fin \quad---\quad Mot : trop grossier
    \quad$\Rightarrow$\quad Y a-t-il un juste milieu ?
  \end{alertblock}
\end{frame}

\begin{frame}{Tokenisation (2/2) --- Subword : Byte Pair Encoding}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{block}{Principe du BPE}
        \begin{enumerate}
          \item Initialiser : chaque caractère est un token
          \item Compter les \textbf{paires} adjacentes les plus fréquentes
          \item Fusionner la paire la plus fréquente $\to$ nouveau token
          \item Répéter jusqu'à la taille de vocabulaire cible
        \end{enumerate}
      \end{block}
      \smallskip
      \begin{block}{Résultat}
        Mots \textbf{fréquents} conservés entiers.\\
        Mots \textbf{rares} découpés en sous-mots significatifs.
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{exampleblock}{Exemple}
        \centering
        \vspace{4pt}
        \small «~inexplicablement~»\\[6pt]
        $\downarrow$\\[4pt]
        \texttt{\small[in \,|\, explicable \,|\, ment]}
        \vspace{4pt}
      \end{exampleblock}
      \smallskip
      \begin{alertblock}{Vocabulaires modernes}
        \begin{itemize}
          \item GPT-2 : 50\,257 tokens (BPE)
          \item BERT : 30\,522 tokens (WordPiece)
          \item Couvrent \textbf{n'importe quel} texte
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

      \begin{alertblock}{Les 3 fléaux du One-Hot}
        \begin{enumerate}
          \item \textbf{Creusité} : $V = 50\,000$ valeurs, un seul \texttt{1}
          \item \textbf{Orthogonalité} :
                $\mathbf{v}_{\text{chat}} \cdot \mathbf{v}_{\text{chien}} = 0$ ---

          \item \textbf{Dimensionnalité} : taille vecteur $=$ taille vocabulaire
        \end{enumerate}
      \end{alertblock}
      \smallskip
      \begin{exampleblock}{La solution : embedding dense}
        Représenter chaque token par un vecteur \textbf{dense}
        de petite dimension $d \ll V$\,:
        \[
          d \in \{64,\; 128,\; 256,\; 768,\; \ldots\}
        \]
      \end{exampleblock}

\section{Word2Vec \& propriétés}
% ============================================================

\begin{frame}{Apprendre l'espace : Word2Vec}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{L'hypothèse distributionnelle}
        «~On connaît un mot par ses voisins.~»\\[6pt]
        \textit{«~le \underline{chat} dort sur le tapis~»}\\[2pt]
        $\Rightarrow$ «~chat~» cooccure avec : dormir, tapis, \ldots
      \end{block}
      \smallskip
      \begin{block}{Méthode Skip-gram}
        \begin{enumerate}
          \item Fenêtre glissante (taille $k$) sur le corpus
          \item Réseau : prédire les mots \textbf{contexte} à partir
                du mot \textbf{cible}
          \item Matrice de poids $\W_{e} \in \R^{V \times d}$
                $=$ dictionnaire d'embeddings
          \item[\,$\star$] On \textbf{jette la prédiction} et ne
                conserve que $\W_{e}$
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \begin{tikzpicture}[font=\scriptsize,
        nd/.style={draw, rounded corners=3pt, minimum width=0.9cm,

\begin{frame}{Word2Vec : CBOW vs Skip-gram}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{CBOW \small(Continuous Bag of Words)}
        \textbf{Entrée} : mots contexte (moyennés)\\
        \textbf{Sortie} : mot cible\\[4pt]
        \centering\small
        $[\text{le},\, \text{dort},\, \text{sur}] \;\to\; \texttt{chat}$
        \begin{itemize}
          \item Plus rapide à entraîner
          \item Meilleur sur grands corpus
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Skip-gram}
        \textbf{Entrée} : mot cible\\
        \textbf{Sortie} : mots contexte\\[4pt]
        \centering\small
        $\texttt{chat} \;\to\; [\text{le},\, \text{dort},\, \text{sur}]$
        \begin{itemize}
          \item Meilleur sur petits corpus
          \item Plus performant sur mots rares
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \bigskip
  \begin{exampleblock}{Les deux apprennent la même chose}
    La tâche de prédiction est un \textbf{prétexte} : ce qui compte, c'est la matrice
    $\W_e \in \R^{V \times d}$ obtenue après entraînement.
    \textbf{Skip-gram} est le plus utilisé en recherche.
  \end{exampleblock}
\end{frame}

\begin{frame}{Entraîner Word2Vec efficacement : Negative Sampling}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{alertblock}{Le problème de la softmax}
        \vspace{-2pt}
        \[
          P(w \mid c) = \frac{e^{\mathbf{u}_w \cdot \mathbf{v}_c}}{\displaystyle\sum_{i=1}^{V} e^{\mathbf{u}_i \cdot \mathbf{v}_c}}
        \]
        \vspace{-4pt}
        $V = 50\,000$ : \textbf{calcul prohibitif} à chaque étape.
      \end{alertblock}
      \begin{block}{Solution}
        \small Pour chaque paire $(w^+, c)$ positive :
        \begin{itemize}
          \item tirer $k$ mots négatifs $w_1^-, \ldots, w_k^-$ au hasard
          \item maximiser $\sigma(\mathbf{u}_{w^+} \cdot \mathbf{v}_c)$
          \item minimiser $\sigma(\mathbf{u}_{w_i^-} \cdot \mathbf{v}_c)$
        \end{itemize}
        Mise à jour sur $k{+}1$ mots seulement. $k = 5$--$20$.
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Exemple}
        \small\textbf{Corpus} : «~le chat dort sur le tapis~»\\[3pt]
        Paire positive : (\emph{chat}, \emph{dort})\\[2pt]
        Paires négatives (aléatoires) :\\[1pt]
        (\emph{chat}, \emph{table}),
        (\emph{chat}, \emph{voiture})\ldots\\[3pt]
        Gain de vitesse $\approx \times\,V / k$.
      \end{exampleblock}
      \vspace{2pt}
      \begin{block}{Tirage pondéré}
        \small Probabilité $\propto f(w)^{3/4}$ :\\
        favorise légèrement les mots rares.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{GloVe : co-occurrence globale}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Idée centrale}
        Word2Vec : fenêtre glissante \textbf{locale}.\\[4pt]
        GloVe exploite la matrice de co-occurrence \textbf{globale} :
        \[
          \X_{ij} = \text{nb de fois que } j \text{ apparaît près de } i
        \]
        Objectif : apprendre $\mathbf{u}_i, \mathbf{v}_j$ tels que :
        \[
          \mathbf{u}_i \cdot \mathbf{v}_j + b_i + b_j \;\approx\; \log \X_{ij}
        \]
        $b_i, b_j$ : biais scalaires par mot (fréquence de base).\\
        Les vecteurs encodent les \textbf{ratios} de co-occurrence.
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Avantages}
        \begin{itemize}
          \item Entraînement très rapide (une passe sur $X$)
          \item Capture les statistiques globales du corpus
          \item Analogies vectorielles aussi bonnes que Word2Vec
        \end{itemize}
      \end{exampleblock}
      \vspace{4pt}
      \begin{block}{Vecteurs pré-entraînés}
        GloVe-6B (Wikipedia + Gigaword) :\\
        400\,000 mots, $d \in \{50, 100, 200, 300\}$.\\
        Disponibles publiquement.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Algèbre de mots \& similarité cosinus}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{exampleblock}{Propriété : analogies vectorielles}
        Les relations sémantiques $=$ translations dans $\R^d$ :
        \[
          \mathbf{v}_{\text{roi}} - \mathbf{v}_{\text{homme}} + \mathbf{v}_{\text{femme}}
          \;\approx\; \mathbf{v}_{\text{reine}}
        \]
        \[
          \mathbf{v}_{\text{Paris}} - \mathbf{v}_{\text{France}} + \mathbf{v}_{\text{Allemagne}}
          \;\approx\; \mathbf{v}_{\text{Berlin}}
        \]
      \end{exampleblock}
      \smallskip
      \begin{block}{Mesure de proximité : similarité cosinus}
        \[
          \cos(\theta) = \frac{\mathbf{a} \cdot \mathbf{b}}{\norm{\mathbf{a}}\,\norm{\mathbf{b}}}
          \;\in\; [-1,\;1]
        \]
        \vspace{-6pt}
        \begin{center}
          \footnotesize
          $1$ : synonymes \quad $0$ : sans rapport \quad $-1$ : antonymes
        \end{center}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \begin{tikzpicture}[font=\tiny, scale=1.15]
        \draw[->, gray!60] (-0.2,0) -- (2.6,0) node[right] {dim 1};

\section{Embeddings en pratique}
% ============================================================

\begin{frame}[fragile]{Embeddings en PyTorch : \texttt{nn.Embedding}}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
\begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize]
import torch, torch.nn as nn

# Matrice W_e : 10 000 mots, dim 128
emb = nn.Embedding(num_embeddings=10_000,
                   embedding_dim=128)

# Indices de tokens  [B=2, T=3]
idx = torch.tensor([[  3, 47, 892],
                    [ 12,  0,   1]])

x = emb(idx)  # [2, 3, 128] -- lookup differentiable

# --- Embeddings pre-entraines (GloVe / Word2Vec) ---
W = torch.FloatTensor(glove_matrix)  # [V, d]
emb = nn.Embedding.from_pretrained(W, freeze=False)
\end{lstlisting}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{\texttt{nn.Embedding} en bref}
        \begin{itemize}
          \item Matrice $\W_e \in \R^{V \times d}$ apprise par gradient
          \item \texttt{emb(idx)} : simple \textbf{lookup} de lignes
          \item Opération \textbf{différentiable}
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{exampleblock}{\texttt{freeze=False}}
        Initialiser avec GloVe / Word2Vec,
        puis continuer l'entraînement :
        c'est du \textbf{transfer learning} au niveau mot.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Réduction de dimension : simplifier le modèle aval}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Au-delà de la visualisation}
        Embeddings $\in \R^{768}$ : espace de haute dimension.\\[4pt]
        Avec \textbf{PCA} (ou UMAP) : $\R^{768} \to \R^{50}$\\[4pt]
        $\Rightarrow$ un classifieur \textbf{simple} suffit souvent :
        \begin{itemize}
          \item Régression logistique
          \item SVM
          \item $k$-NN
        \end{itemize}
      \end{block}
      \smallskip
      \begin{exampleblock}{Pipeline typique}
        \centering\small
        \texttt{BERT / Word2Vec}
        $\;\to\;$ \texttt{PCA}
        $\;\to\;$ \texttt{LogisticRegression}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Avantages}
        \begin{itemize}
          \item Entraînement \textbf{rapide}
          \item Moins de données nécessaires
        \end{itemize}
      \end{exampleblock}
      \vspace{4pt}
      \begin{alertblock}{Limite : surapprentissage}
        En haute dimension, le classifieur mémorise\\
        les données d'entraînement (\textbf{overfitting}).\\[3pt]
        La réduction de dimension \textbf{régularise}\\
        implicitement le modèle aval.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Limites des embeddings statiques}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{alertblock}{Le problème : la polysémie}
        \begin{itemize}
          \item «~Je dépose de l'argent à la \textbf{banque}.~»
          \item «~Je pêche au bord de la \textbf{banque}.~»
        \end{itemize}
        \vspace{4pt}
        Dans Word2Vec / GloVe, «~banque~» possède
        \textbf{un seul vecteur}, mélangeant les deux sens.
      \end{alertblock}
      \smallskip
      \begin{alertblock}{Autres limites}
        \begin{itemize}
          \item Ignorent l'ordre des mots
          \item Embeddings \textbf{figés} après entraînement
          \item Biais hérités des données d'entraînement
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{La solution : embeddings \textbf{contextuels}}
        Le vecteur d'un mot doit dépendre de son \textbf{contexte}.
        \vspace{4pt}
        \begin{center}
          \begin{tikzpicture}[font=\tiny,
            box/.style={draw, rounded corners=3pt, align=center,
                        minimum width=2.8cm, minimum height=0.6cm}]
            \node[box, fill=jedy_alert!15, draw=jedy_alert] (b1) at (0,0)
              {banque (finance)};
            \node[box, fill=jedy_blue!15, draw=jedy_blue] (b2) at (0,-1.1)
              {banque (rivi\`ere)};
            \draw[->, jedy_alert, thick] (b1.east) -- (2.0, 0.0)
              node[right] {$\mathbf{v}_1$};
            \draw[->, jedy_blue, thick]  (b2.east) -- (2.0,-1.1)
              node[right] {$\mathbf{v}_2$};
            \node[font=\tiny\bfseries, color=jedy_example] at (2.2,-0.55)
              {$\mathbf{v}_1 \neq \mathbf{v}_2$};
          \end{tikzpicture}
        \end{center}
      \end{block}
      \vspace{2pt}
      \begin{exampleblock}{J4-PM $\;\to\;$ mécanisme d'attention}
        L'attention permet de construire des embeddings
        \textbf{dynamiques} selon le contexte.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Encodage positionnel : vers les Transformers}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{alertblock}{Le problème}
        Les embeddings sont des \textbf{sacs de mots} :\\[3pt]
        «~le chat mange la souris~» $\equiv$ «~la souris mange le chat~»\\[4pt]
        $\Rightarrow$ Il faut encoder la \textbf{position} de chaque token.
      \end{alertblock}
      \smallskip
      \begin{block}{Encodage sinusoïdal (Vaswani 2017)}
        \small
        \begin{align*}
          PE_{(pos,\,2i)}   &= \sin\!\Bigl(\tfrac{pos}{10000^{2i/d}}\Bigr)\\[2pt]
          PE_{(pos,\,2i+1)} &= \cos\!\Bigl(\tfrac{pos}{10000^{2i/d}}\Bigr)
        \end{align*}
        \normalsize
        On \textbf{additionne} à l'embedding : $\mathbf{x}_t \leftarrow \mathbf{x}_t + \mathbf{pe}_t$
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Intuition}
        \begin{itemize}
          \item Chaque position $\to$ vecteur unique $\in \R^d$
          \item Fréquences variées $\to$ positions proches
                $=$ vecteurs similaires
          \item \textbf{Aucun paramètre} supplémentaire
        \end{itemize}
      \end{exampleblock}
      \vspace{4pt}
      \begin{block}{J4-PM $\;\to\;$ Transformers}
        L'encodage positionnel est la \textbf{première brique}
        de l'architecture Transformer.\\[3pt]
        Variante : PE appris (BERT, GPT).
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Récapitulatif --- Ce qu'on a vu}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Tokenisation \& représentations}
        \begin{itemize}\small
          \item Niveaux : caractère / mot / sous-mot (BPE)
          \item One-Hot : creusité, orthogonalité, grande dim.
          \item Embedding dense $\in \R^d$ : pont discret $\to$ continu
          \item Espace sémantique : proximité $=$ sens partagé
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{block}{En pratique}
        \begin{itemize}\small
          \item \texttt{nn.Embedding} : lookup différentiable
          \item \texttt{from\_pretrained} + \texttt{freeze}
          \item Encodage positionnel $\to$ Transformers (J4-PM)
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Word2Vec \& GloVe}
        \begin{itemize}\small
          \item Hypothèse distributionnelle
          \item CBOW vs Skip-gram (tâche prétexte)
          \item Negative sampling : efficacité sur grand $V$
          \item GloVe : co-occurrence globale
          \item Analogies vectorielles, similarité cosinus
          \item t-SNE : visualisation des clusters
          \item Polysémie $\to$ embeddings contextuels (J4-PM)
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
