\begin{frame}{Pourquoi passer à PyTorch ?}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Limite du «~fait main~» (J3-AM)}
        \begin{itemize}
          \item Notre classe \code{Value} est \textbf{pédagogique} mais lente
          \item Python pur : quelques milliers d'opérations/s
          \item Un vrai réseau : \textbf{milliards} d'opérations par forward
        \end{itemize}
      \end{alertblock}
      \smallskip
      \begin{exampleblock}{Même principe, autre échelle}
        Même graphe de calcul, même \code{backward()}\\
        $\Rightarrow$ mais sur des \textbf{tenseurs} en C++/CUDA
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{La solution industrielle}
        \begin{itemize}
          \item \textbf{Calcul vectorisé} : matrices entières en une opération
          \item \textbf{Backend C++/CUDA} : parallélisme GPU
          \item \textbf{Écosystème riche} :
            \begin{itemize}
              \item \code{torchvision} — Vision par ordinateur
              \item \code{torchaudio} — Audio
              \item \code{transformers} — NLP / LLM
            \end{itemize}
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}


\begin{frame}[fragile]{Le Concept de Tenseur (\texttt{torch.Tensor})}
  \begin{center}
    \Large\bfseries\color{jedy_blue}
    \texttt{torch.Tensor} $=$ \texttt{np.ndarray} $+$ autograd
  \end{center}
  \medskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{\texttt{np.ndarray}}
        \begin{lstlisting}[style=pythonstyle]
import numpy as np

x = np.array([[1.0, 2.0],
               [3.0, 4.0]])
y = x ** 2
print(y.sum())  # 30.0
# Pas de gradient !
        \end{lstlisting}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{\texttt{torch.Tensor}}
        \begin{lstlisting}[style=pythonstyle]
import torch

x = torch.tensor([[1.0, 2.0],
                   [3.0, 4.0]],
                  requires_grad=True)
y = x ** 2
y.sum().backward()
print(x.grad)  # 2*x
        \end{lstlisting}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Nouveaux concepts : \texttt{requires\_grad} et \texttt{device}}
  \begin{block}{\texttt{requires\_grad} : suivi du calcul}
    \begin{columns}[T]
      \begin{column}{0.44\textwidth}
        \begin{itemize}
          \item Marque un tenseur comme \textbf{variable à optimiser}
          \item PyTorch enregistre toutes les opérations le concernant
          \item \code{.backward()} remonte le graphe et calcule \code{.grad}
        \end{itemize}
      \end{column}
      \begin{column}{0.52\textwidth}
        \begin{lstlisting}[style=pythonstyle]
x = torch.tensor([2.0], requires_grad=True)
y = x ** 3
y.backward()
print(x.grad)  # 3*x^2 = 12.0
        \end{lstlisting}
      \end{column}
    \end{columns}
  \end{block}
  \medskip
  \begin{block}{\texttt{device} : CPU ou GPU}
    \begin{columns}[T]
      \begin{column}{0.44\textwidth}
        \begin{itemize}
          \item Les tenseurs vivent sur un \textbf{device}
          \item Les opérations exigent le \textbf{même device}
          \item \code{.to(device)} déplace un tenseur ou un modèle
        \end{itemize}
      \end{column}
      \begin{column}{0.52\textwidth}
        \begin{lstlisting}[style=pythonstyle]
device = (
    "cuda" if torch.cuda.is_available()
    else "cpu"
)
x = x.to(device)
model = model.to(device)
        \end{lstlisting}
      \end{column}
    \end{columns}
  \end{block}
\end{frame}

\begin{frame}[fragile]{La Philosophie \texttt{torch.nn}}
  \begin{columns}[T]
    \begin{column}{0.38\textwidth}
      \begin{block}{Le \texttt{Module} : brique de base}
        \begin{itemize}\small
          \item On ne gère \textbf{plus} $\W$ et $\bm{b}$ manuellement
          \item \code{nn.Linear}, \code{nn.Conv2d}... gèrent leurs paramètres
          \item \code{\_\_init\_\_} : définit les couches
          \item \code{forward()} : flux de données
        \end{itemize}
      \end{block}
      \smallskip
      \begin{exampleblock}{Avantage}
        \small\code{model.parameters()} retourne\\
        \textbf{tous} les poids $\Rightarrow$ prêts pour l'optimiseur
      \end{exampleblock}
    \end{column}
    \begin{column}{0.58\textwidth}
      \begin{lstlisting}[style=pythonstyle]
import torch.nn as nn
import torch.nn.functional as F

class MLP(nn.Module):
    def __init__(self, n_in, n_h, n_out):
        super().__init__()
        self.fc1 = nn.Linear(n_in, n_h)
        self.fc2 = nn.Linear(n_h, n_out)

    def forward(self, x):
        x = F.relu(self.fc1(x))
        return self.fc2(x)

model = MLP(784, 128, 10)
# Nombre de parametres :
n = sum(p.numel() for p in model.parameters())
print(f"{n} parametres")  # 101770
      \end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}


\begin{frame}[fragile]{Optimisation : \texttt{torch.optim}}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Plus de \code{w -= lr*grad} manuel}
        \begin{itemize}
          \item \textbf{SGD} : descente de gradient classique
          \item \textbf{Adam} : adaptatif, converge rapidement
          \item \textbf{RMSprop} : stable pour les RNN
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Interface unifiée}
        Peu importe l'optimiseur, l'API est toujours la même :
        \code{opt.zero\_grad()} / \code{opt.step()}
      \end{exampleblock}
    \end{column}
  \end{columns}
  \medskip
  \begin{lstlisting}[style=pythonstyle]
import torch.optim as optim

# SGD classique
opt = optim.SGD(model.parameters(), lr=1e-2, momentum=0.9)

# Adam (defaut recommande)
opt = optim.Adam(model.parameters(), lr=1e-3)

# Meme interface pour tous :
opt.zero_grad()
loss.backward()
opt.step()
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{La Training Loop Standard}
  \begin{columns}[T]
    \begin{column}{0.37\textwidth}
      \begin{block}{Les 5 étapes rituelles}
        \begin{enumerate}\small
          \item \code{zero\_grad()} : efface les anciens gradients
          \item \code{model(x)} : forward pass
          \item \code{criterion(out, y)} : calcul de la perte
          \item \code{loss.backward()} : autograd
          \item \code{optimizer.step()} : mise à jour des poids
        \end{enumerate}
      \end{block}
      \smallskip
      \begin{alertblock}{\small Piège classique}
        \small Oublier \code{zero\_grad()} $\Rightarrow$ gradients \textbf{accumulés} $\Rightarrow$ divergence
      \end{alertblock}
    \end{column}
    \begin{column}{0.59\textwidth}
      \begin{lstlisting}[style=pythonstyle]
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters())

for epoch in range(n_epochs):
    for X, y in train_loader:
        # 1. Zero gradients
        optimizer.zero_grad()

        # 2. Forward pass
        output = model(X)

        # 3. Compute loss
        loss = criterion(output, y)

        # 4. Backward (autograd)
        loss.backward()

        # 5. Update weights
        optimizer.step()
      \end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Pourquoi le MLP Échoue sur les Images ?}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Problème 1 : Explosion des paramètres}
        \begin{itemize}
          \item Image $1024 \times 1024 \Rightarrow 10^6$ entrées
          \item Couche cachée de 512 neurones
          \item $10^6 \times 512 = \mathbf{5 \times 10^8}$ poids
          \item Rien que pour la \textbf{première couche} !
        \end{itemize}
      \end{alertblock}
      \smallskip
      \begin{alertblock}{Problème 2 : Perte de topologie}
        \begin{itemize}
          \item Flatten $\Rightarrow$ perte de la \textbf{structure 2D}
          \item Un décalage d'1 pixel change tout le vecteur
          \item Le réseau ne voit pas la \textbf{proximité spatiale}
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Solution : CNN}
        \begin{itemize}
          \item \textbf{Localité} : chaque neurone voit une petite zone
          \item \textbf{Partage des poids} : même filtre sur toute l'image
          \item \textbf{Hiérarchie} : bords $\to$ formes $\to$ objets
        \end{itemize}
      \end{exampleblock}
      \smallskip
      \begin{center}
        \renewcommand{\arraystretch}{1.3}
        \begin{tabular}{lrr}
          \toprule
           & \textbf{MLP} & \textbf{CNN} \\
          \midrule
          Params ($1^{\text{re}}$ couche) & $500\text{M}$ & $\sim 1\text{K}$ \\
          Invariance translation & Non & Oui \\
          Localité spatiale & Non & Oui \\
          \bottomrule
        \end{tabular}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Stride, Padding et Taille de Sortie}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{Stride $S$ : pas du filtre}
        \begin{itemize}\small
          \item $S=1$ : filtre avance pixel par pixel (défaut)
          \item $S=2$ : saute un pixel $\Rightarrow$ sortie 2$\times$ plus petite
        \end{itemize}
      \end{block}
      \vspace{-4pt}
      \begin{block}{Padding $P$ : zéros autour}
        \begin{itemize}\small
          \item $P=0$ : sortie rétrécie de $K{-}1$ de chaque côté
          \item $P=\lfloor K/2 \rfloor$ : \textbf{same padding}, taille préservée
        \end{itemize}
      \end{block}
      \vspace{-4pt}
      \begin{exampleblock}{Taille de sortie}
        \centering $\displaystyle H_{out} = \left\lfloor \frac{H_{in} - K + 2P}{S} \right\rfloor + 1$
      \end{exampleblock}
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{lstlisting}[style=pythonstyle, basicstyle=\ttfamily\scriptsize]
nn.Conv2d(
    in_channels  = 1,   # canaux entree
    out_channels = 32,  # nb de filtres
    kernel_size  = 3,   # K x K
    stride       = 1,   # S
    padding      = 1,   # P -> same padding
)
# H_in=28, K=3, P=1, S=1
# H_out = (28 - 3 + 2) / 1 + 1 = 28
      \end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Le Pooling : Réduction et Invariance}
  \begin{columns}[T]
    \begin{column}{0.44\textwidth}
      \begin{block}{Max Pooling $2 \times 2$}
        On garde la valeur \textbf{maximale} dans chaque zone $2\times2$
      \end{block}
      \smallskip
      \begin{block}{Objectifs}
        \begin{enumerate}
          \item \textbf{Réduire} la résolution spatiale (moins de calculs)
          \item \textbf{Invariance} aux petites translations
          \item \textbf{Élargir} le champ de vision des couches suivantes
        \end{enumerate}
      \end{block}
      \smallskip
      \begin{exampleblock}{Résultat}
        Pool $2\!\times\!2$ : $H\!\times\!W \;\to\; \frac{H}{2}\!\times\!\frac{W}{2}$\\
        $\Rightarrow$ 4$\times$ moins de valeurs
      \end{exampleblock}
    \end{column}
    \begin{column}{0.50\textwidth}
      \centering
      \begin{tikzpicture}[scale=0.90]
        % --- Input 4x4 hardcoded ---
        % Row 3 (top): 1 3 2 4
        % Row 2:       5 6 1 2
        % Row 1:       0 2 7 8
        % Row 0 (bot): 3 1 4 6

        % Zone bleue haut-gauche
        \draw[jedy_blue, thick, fill=jedy_blue!15] (0,2) rectangle (2,4);

\begin{frame}[fragile]{CNN en PyTorch : Schéma d'Architecture}
  \begin{lstlisting}[style=pythonstyle, basicstyle=\ttfamily\scriptsize]
class CNN(nn.Module):
    def __init__(self):
        super().__init__()
        # --- Extraction de features ---
        self.conv1 = nn.Conv2d(..., ..., kernel_size=...)
        self.conv2 = nn.Conv2d(..., ..., kernel_size=...)
        self.pool  = nn.MaxPool2d(kernel_size=2)
        # --- Tete de classification ---
        self.fc1   = nn.Linear(..., ...)
        self.fc2   = nn.Linear(..., n_classes)

    def forward(self, x):      # x : [B, C, H, W]
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.flatten(start_dim=1)  # [B, ...]
        x = F.relu(self.fc1(x))
        return self.fc2(x)     # logits [B, n_classes]
  \end{lstlisting}
  \begin{block}{Lecture du flux}
    \footnotesize
    \textbf{1.}~\code{conv1}+\code{ReLU}+\code{pool} $\to$ feature maps, résolution $\div 2$ \enspace
    \textbf{2.}~\code{conv2}+\code{ReLU}+\code{pool} $\to$ features abstraites, $\div 2$ \enspace
    \textbf{3.}~\code{flatten} : 3D $\to$ 1D \\[2pt]
    \textbf{4.}~\code{fc1}+\code{ReLU} : couche dense \enspace
    \textbf{5.}~\code{fc2} : logits $\in \R^{n\_classes}$
  \end{block}
\end{frame}

      \begin{block}{PyTorch}
        \begin{itemize}
          \item \code{Tensor} = \code{ndarray} + autograd (\code{requires\_grad}, \code{device})
          \item \code{nn.Module} : couches, \code{forward()}, \code{parameters()}
          \item \code{torch.optim} : SGD, Adam — même interface
          \item \code{Dataset} / \code{DataLoader} : batching, shuffling
          \item Training loop : \code{zero\_grad} / forward / loss / \code{backward} / \code{step}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{CNN}
        \begin{itemize}
          \item MLP : explosion des paramètres, perte de topologie
          \item Convolution : localité, partage des poids, kernel $K\!\times\!K$
          \item Stride $S$, padding $P$ — formule $H_{out}$
          \item $N$ filtres $\to$ $N$ feature maps $[B, N, H', W']$
          \item Pooling : réduction $\times 2$, invariance aux translations
          \item Architecture : Conv+ReLU+Pool $\to$ Flatten $\to$ FC
        \end{itemize}
      \end{block}
    \end{column}
